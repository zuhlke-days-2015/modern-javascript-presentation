Angular 1.x Component

<li movie-tile imageurl="img/image1.png"></li>
angular
	.module('movieApp', [])
	.directive('movie-tile', function() {
		return {
			scope: {},
			bindToController: {
				imageurl: '@'
			}
			controller: function () { },
			controllerAs: 'ctrl',
			template: '<img src="{{ctrl.imageurl}}" />'
		};
	});

ES6 Component

<li is="movie-tile" imageurl="img/image1.png"></li>
class MovieTile extends HTMLLIElement {
	createdCallback() {
		this.render(this.attributes);
	}
	
	render(attrs) {
		this.innerHTML = `
			<img src="${attrs.imageurl.value}" />
		`;
	}
}
	
export default document.registerElement('movie-tile', {
	prototype: MovieTile.prototype,
	extends: 'li'
});
